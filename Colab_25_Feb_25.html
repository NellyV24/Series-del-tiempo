import yfinance as yf
import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller

# Fetch historical stock data for NKE from November 29, 2019, to today
start_date = "2019-11-29"
end_date = "2025-02-25"
ticker = "NKE"
stock_data = yf.download(ticker, start=start_date, end=end_date)

# Perform random walk test (using ADF test)
def adf_test(timeseries):
    adf_result = adfuller(timeseries, autolag='AIC')
    print(f"ADF Statistic: {adf_result[0]}")
    print(f"p-value: {adf_result[1]}")
    print("Critical Values:")
    for key, value in adf_result[4].items():
        print(f"\t{key}: {value}")

print("Performing ADF test on Close prices...")
adf_test(stock_data['Close'])

# Calculate moving averages
stock_data['MA9'] = stock_data['Close'].rolling(window=9).mean()
stock_data['MA30'] = stock_data['Close'].rolling(window=30).mean()

# Plot the stock data along with moving averages
plt.figure(figsize=(14, 7))
plt.plot(stock_data['Close'], label='Close Price')
plt.plot(stock_data['MA9'], label='9-Day Moving Average')
plt.plot(stock_data['MA30'], label='30-Day Moving Average')
plt.title(f'{ticker} Stock Price and Moving Averages')
plt.xlabel('Date')
plt.ylabel('Price')
plt.legend()
plt.show()
