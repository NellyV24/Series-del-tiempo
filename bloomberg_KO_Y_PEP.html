# Function for unit root tests with interpretation
def unit_root_tests(series, name):
    print(f"\nUnit Root Tests for {name} (3-min, 3 days):")

    # ADF Test
    adf_result = adfuller(series)
    print("ADF Test:")
    print(f'ADF Statistic: {adf_result[0]:.4f}')
    print(f'p-value: {adf_result[1]:.4f}')
    print(f'Critical Values: {adf_result[4]}')
    if adf_result[1] < 0.05:
        print(f"  - Stationary at 5% level")
    else:
        print(f"  - Non-stationary at 5% level")

    # KPSS Test
    kpss_result = kpss(series, regression='c')
    print("\nKPSS Test:")
    print(f'KPSS Statistic: {kpss_result[0]:.4f}')
    print(f'p-value: {kpss_result[1]:.4f}')
    print(f'Critical Values: {kpss_result[3]}')
    if kpss_result[1] < 0.05:
        print(f"  - Non-stationary at 5% level")
    else:
        print(f"  - Stationary at 5% level")

# Perform unit root tests
unit_root_tests(ko_close, "KO")
unit_root_tests(pep_close, "PEP")

# Function to find best ARMA model (d=0, from first version)
def find_best_arma(series, name, p_range=3, q_range=3):
    best_aic = float('inf')
    best_order = None
    for p in range(p_range + 1):
        for q in range(q_range + 1):
            try:
                model = ARIMA(series, order=(p, 0, q))
                results = model.fit()
                if results.aic < best_aic:
                    best_aic = results.aic
                    best_order = (p, 0, q)
            except:
                continue
    print(f"\nBest ARMA model for {name} (3-min, 3 days): Order={best_order}, AIC={best_aic:.2f}")
    return best_order

# Find and fit ARMA models on differenced series

def johansen_coint_test(df):
    result = coint_johansen(df, det_order=0, k_ar_diff=1)
    print("\nJohansen Cointegration Test (3-min, 3 days):")
    print(f"Trace statistic: {result.lr1}")
    print(f"Critical values (90%, 95%, 99%): {result.cvt}")
    for i in range(len(result.lr1)):
        if result.lr1[i] > result.cvt[i, 1]:
            print(f"  - r = {i}: Cointegration at 95% confidence level")
        else:
            print(f"  - r = {i}: No cointegration at 95% confidence level")

# Detailed forecast plot with confidence intervals
def plot_forecast(model, series, name, steps=30):
    forecast_obj = model.get_forecast(steps=steps)
    forecast = forecast_obj.predicted_mean
    conf_int = forecast_obj.conf_int()
    forecast_index = range(len(series), len(series) + steps)

    # Reconstruct to original scale
    forecast_cumsum = series.iloc[-1] + forecast.cumsum()
    conf_int_cumsum_lower = series.iloc[-1] + conf_int.iloc[:, 0].cumsum()
    conf_int_cumsum_upper = series.iloc[-1] + conf_int.iloc[:, 1].cumsum()

    plt.figure(figsize=(12, 6))
    plt.plot(series, label=f'{name} (3-min, 3 days)')
    plt.plot(forecast_index, forecast_cumsum, label='Forecast', color='red')
    plt.fill_between(forecast_index, conf_int_cumsum_lower, conf_int_cumsum_upper,
                     color='pink', alpha=0.3, label='95% Confidence Interval')
    plt.title(f'{name} Price Forecast (3-min intervals)')
    plt.legend()
    plt.show()
